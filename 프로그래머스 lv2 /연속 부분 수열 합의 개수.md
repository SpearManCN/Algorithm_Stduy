import java.util.*;
class Solution {
    public int solution(int[] a) {
        Set<Integer> set = new HashSet<>();
        int len = a.length;
        for(int i=0; i<a.length; i++){
            int tmp = 0;
            int no = 0;
            for(int j=0; j<a.length; j++){ 
                no = (i+j)%len;
                tmp+=a[no];
                set.add(tmp);
            }
        }
        return set.size();
    }
}
