class Solution { 
    public int[] solution(int n, long left, long right) {
        int gap = (int)(right - left);
        int startRow = (int)(left/n)+1;
        int startCol = (int)((left+1)%n); 
        int[] pool = new int[gap+1];
        for(int i=0; i<=gap; i++){
            if(startCol==0){ 
                pool[i]=n;
                startRow++;
                startCol++;
                startCol%=n;
            }else if(startRow>=startCol){
                pool[i]=startRow;
                startCol++;
                startCol%=n;
            }else{
                pool[i]=startCol;
                startCol++;
                startCol%=n;
            }
        }
        return pool;
        
    }
}
