import java.util.*;
class Solution {
    public int[] solution(int n, String[] words) {
        Set<String> set = new HashSet<>();
        char last = words[0].charAt(words[0].length()-1);
        set.add(words[0]);
        int failNo = 0;
        for(int i=1; i<words.length; i++){
            int before = set.size();
            set.add(words[i]);
            
            if(last != words[i].charAt(0)){
                failNo = i;
                break;
            }
            
            if(before == set.size()){
                failNo = i;
                break;
            }
            
            last = words[i].charAt(words[i].length()-1);
        }
        int[] result = {0,0};
        if(failNo != 0 ){
            result[0] = (failNo)%n+1;
            result[1] = ((failNo)/n)+1;
        }
        return result;
    }
}
