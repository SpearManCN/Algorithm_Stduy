class Solution {
    public int[] solution(String s) {
        int length = s.length();
        int count = 0;
        int loop = 0;
        while(length!=1){
            s=s.replaceAll("0","");
            count+=length-s.length();
            length = s.length();
            s = Integer.toBinaryString(length);
            length = s.length();
            loop++;
        }
        int[] result = {loop, count};
        return result;
    }
}
