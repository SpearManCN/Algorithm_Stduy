import java.util.*;
class Solution {
    public int[] solution(String s) {
        int[] alpha = new int[26];
        Arrays.fill(alpha, -1);
        char[] tmp = s.toCharArray();
        int tmpLength = tmp.length;
        int[] result = new int[tmp.length];
        for(int i = 0 ; i < tmpLength; i++){
            int theVal =  tmp[i] - 'a';
            if (alpha[theVal]==-1){
                result[i] = -1;
                alpha[theVal]=i;
            }else{
                result[i] = i - alpha[theVal];
                alpha[theVal] = i;
            }
        }
        return result;
    }
}
